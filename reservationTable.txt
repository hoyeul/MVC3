create table rev(
rev_num varchar2(4) primary key,
rev_name varchar2(20) not null,
rev_date varchar2(20) not null,
rev_time varchar2(10) not null unique,
rev_location varchar2(20) not null,
rev_program varchar2(30) not null
);

rev_num : 예약번호
rev_name : 이름
rev_date : 날짜
rev_time : 시간
rev_location : 위치
rev_program : 프로그램
